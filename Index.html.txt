<script>

let products = JSON.parse(localStorage.getItem("products")) || [

{
name:"Nuggets",
price:500,
image:"https://images.unsplash.com/photo-1604908812857-9f3b3c1d8b4a"
},
{
name:"Chapli Kabab",
price:800,
image:"https://images.unsplash.com/photo-1601050690597-df0568f70950"
}

];

let cart=[];

// SAVE
function save(){
localStorage.setItem("products",JSON.stringify(products));
}

// LOAD PRODUCTS
function loadProducts(){

let container=document.getElementById("products");
let adminList=document.getElementById("adminProducts");

container.innerHTML="";
adminList.innerHTML="";

products.forEach((p,i)=>{

container.innerHTML+=`
<div class="card">
<img src="${p.image}">
<h3>${p.name}</h3>
<div class="price">Rs ${p.price}</div>
<button class="btn" onclick="addToCart(${i})">Add To Cart</button>
</div>
`;

adminList.innerHTML+=`
<div>
${p.name} - Rs ${p.price}
<button onclick="deleteProduct(${i})">Delete</button>
</div>
`;

});

}

// ADD PRODUCT
function addProduct(){

let name=document.getElementById("name").value;
let price=document.getElementById("price").value;
let image=document.getElementById("image").value;

if(!name || !price || !image){
alert("Fill all fields");
return;
}

products.push({name,price,image});
save();
loadProducts();

}

// DELETE
function deleteProduct(i){
products.splice(i,1);
save();
loadProducts();
}

// CART
function addToCart(i){
cart.push(products[i]);
updateCart();
}

function updateCart(){

document.getElementById("count").innerText=cart.length;

let items="";
let total=0;

cart.forEach(item=>{
items+=`<div class="cart-item">
<span>${item.name}</span>
<span>${item.price}</span>
</div>`;
total+=Number(item.price);
});

document.getElementById("cartItems").innerHTML=items;
document.getElementById("total").innerText="Total: Rs "+total;

}

function toggleCart(){
document.getElementById("cart").classList.toggle("active");
}

// ADMIN LOGIN
function adminLogin(){

let pass=prompt("Enter Admin Password");

if(pass==="hanis123"){
document.getElementById("adminPanel").classList.toggle("active");
}else{
alert("Wrong Password");
}

}

// WHATSAPP ORDER
function orderWhatsApp(){

let msg="Order:%0A";

cart.forEach(item=>{
msg+=`${item.name} - Rs ${item.price}%0A`;
});

window.open(`https://wa.me/92XXXXXXXXXX?text=${msg}`);

}

loadProducts();
